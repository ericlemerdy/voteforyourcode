import scala.collection.mutable.ListBuffer

object FooBarQix{
	
	val foobarqix=List(
		("Foo",3),
		("Bar",5),
		("Qix",7)
	)
	
	def main(args:Array[String]){
		(1 to 100)
			.map(map) 		// For each element, map a list with the element followed by the foo-bar-qix items, if applicable
			.map(reduce) 	// Next, remove first element if we have foo-bar-qix
			.foreach(print)	// Finally, just print it
	}
	
	def map(num:Int)={
		var list=new ListBuffer[String]()
		list+=num.toString 
		foobarqix.foreach(test=>if(num%test._2==0) list+=test._1) // Multiples
		num.toString.foreach(c=>foobarqix.foreach(test=>if(c==test._2.toString.head) list+=test._1)) // Content
		list.toList
	}

	def reduce(list:List[String]):List[String]={if(list.size>1) list.drop(1) else list}
	
	def print(list:List[String])={println(list.mkString)}
	
}
