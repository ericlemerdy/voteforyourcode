package org.kocakosm.foobarqix;

/**
 * A FooBarQix number.
 *
 * @author Osman KOCAK
 */
public final class FooBarQixNumber
{
	private static final String FOO = "Foo";
	private static final String BAR = "Bar";
	private static final String QIX = "Qix";

	/**
	 * Returns a {@code FooBarQixNumber} instance representing the
	 * specified {@code int} value.
	 *
	 * @param value an {@code int} value.
	 *
	 * @return a {@code FooBarQixNumber} representing the given value.
	 */
	public static FooBarQixNumber valueOf(int value)
	{
		if (value == 0) {
			return new FooBarQixNumber(0, "0");
		}
		String number = Integer.toString(value);
		StringBuilder sb = new StringBuilder(number.length() * 3);
		if (value % 3 == 0) {
			sb.append(FOO);
		}
		if (value % 5 == 0) {
			sb.append(BAR);
		}
		if (value % 7 == 0) {
			sb.append(QIX);
		}
		for (char c : number.toCharArray()) {
			if (c == '3') {
				sb.append(FOO);
			} else if (c == '5') {
				sb.append(BAR);
			} else if (c == '7') {
				sb.append(QIX);
			}
		}
		return new FooBarQixNumber(
			value, sb.length() == 0 ? number : sb.toString());
	}

	private final int intValue;
	private final String strValue;

	private FooBarQixNumber(int intValue, String strValue)
	{
		this.intValue = intValue;
		this.strValue = strValue;
	}

	/**
	 * Returns the value of this {@code FooBarQixNumber} as an {@code int}.
	 *
	 * @return the numeric value represented by this object.
	 */
	public int intValue()
	{
		return intValue;
	}

	@Override
	public String toString()
	{
		return strValue;
	}

	@Override
	public boolean equals(Object o)
	{
		if (o == this) {
			return true;
		}
		if (!(o instanceof FooBarQixNumber)) {
			return false;
		}
		final FooBarQixNumber number = (FooBarQixNumber) o;
		return intValue == number.intValue;
	}

	@Override
	public int hashCode()
	{
		return intValue;
	}
}
