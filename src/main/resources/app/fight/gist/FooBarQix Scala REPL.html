object FooBarQix {
  val tokens = Map(3->"Foo", 5->"Bar", 7->"Qix")
 
  def present(v:Int) = v.toString.collect { case t => tokens.get( "" + t toInt) } flatten
 
  def dividers(v:Int) = tokens.filter{v % _._1 == 0}.values
 
  def changeMe(v:Int) = { 
   val x = (dividers(v) :: present(v)::Nil flatten) 
   if (x.isEmpty) v.toString else x.mkString
  }
}
 
// Print results:
for ( i <- 1 to 100) println(FooBarQix.changeMe(i))
