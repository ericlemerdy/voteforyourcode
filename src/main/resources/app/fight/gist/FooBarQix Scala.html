object FooBarQix 
{
	def main(args: Array[String]) =
		(1 to 100).foreach { n => println(fooBarQix(n)) }
	
	def fooBarQix(n:Int): String =
		divisors(n) ++ digits(n) match { case "" => n.toString; case fbqn:String => fbqn }
	
	def divisors(n: Int): String =
		fbq.keys.map { key => if (n % key == 0) fbq(key) else "" }.mkString
	
	def digits(n: Int): String =
		n.toString.foldLeft("")((s, c) => s + fbq.getOrElse(c.asDigit, ""))
	
	val fbq = Map(3 -> "Foo", 5 -> "Bar", 7 -> "Qix")
}
